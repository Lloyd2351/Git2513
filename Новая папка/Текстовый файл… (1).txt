cd - поменять дерикторию
pwd - текущая директория
относительный путь начинается с ./
cd ../.. - вверх по иерархии на 2 уровня, ../../.. - на 3 уровня
~ - домашняя директория
ls - посмотреть директорию
mkdir - создать директорию
rmdir - удалить директорию
touch - создать файл
echo - вывод в консоль
> - перенаправить в файл и перезаписать
>> - перенаправить в файл и дозаписать в конец
| - использовать данные для след команды
grep - сортировка []
:wq - выйти из редактора
